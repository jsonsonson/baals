    // Esse observador intercepta qualquer tag script que o Poki tentar injetar 
    // com caminho relativo e força a rota para a raiz real do seu jsDelivr.
    const observer = new MutationObserver((mutations) => {
        mutations.forEach((mutation) => {
            mutation.addedNodes.forEach((node) => {
                if (node.tagName === 'SCRIPT' && node.src) {
                    if (node.src.includes('poki-sdk-core.js') && !node.src.startsWith('http')) {
                        node.src = REPO_URL + 'poki-sdk-core.js';
                    }
                }
            });
        });
    });
    observer.observe(document.documentElement, { childList: true, subtree: true });

    // CORREÇÃO UNIVERSAL DE WEB WORKERS
    const OriginalWorker = window.Worker;
    window.Worker = function(scriptURL) {
        let fullURL = scriptURL;
        
        if (!fullURL.startsWith('http://') && !fullURL.startsWith('https://')) {
            fullURL = REPO_URL + fullURL.replace(/^\//, '');
        }
        
        if (scriptURL.includes('jsdelivr.net') || scriptURL.includes('js/workers/') || scriptURL.includes('worker') || scriptURL.includes('poki')) {
            if (fullURL.includes('?')) {
                fullURL = fullURL.split('?')[0];
            }
            if (fullURL.endsWith('.png')) {
                fullURL = fullURL.replace('.png', '.webp');
            }
            const blobCode = "importScripts('" + fullURL + "');";
            const blob = new Blob([blobCode], { type: 'application/javascript' });
            const blobURL = URL.createObjectURL(blob);
            return new OriginalWorker(blobURL);
        }
        return new OriginalWorker(fullURL);
    };
    </script>

    function iniciarJogo(idioma) {
      console.log("Iniciando motor de rotas universal...");
      
      const URL_BASE_JOGO = "https://cdn.jsdelivr.net/gh/jsonsonson/baals@main/";
      const originalFetch = window.fetch;
      
      window.fetch = async function(...args) {
        let url = args[0];
        
        if (typeof url === 'string') {
          url = decodeURIComponent(url);

          if (url.includes('?')) {
            url = url.split('?')[0];
          }

          if (url.includes("content:/media/external/file/")) {
             url = url.split("content:/media/external/file/")[1];
          }

          // INTERCEPTADOR DE CORREÇÃO DE EXTENSÕES E PASTAS VIA FETCH
          if (!url.startsWith(URL_BASE_JOGO)) {
             let caminhoReal = url;
             
             if (url.includes('poki-sdk-core.js')) {
                 caminhoReal = 'poki-sdk-core.js';
             } else if (url.includes('assets/')) {
                 caminhoReal = 'assets/' + url.split('assets/')[1];
             } else if (url.includes('js/')) {
                 caminhoReal = 'js/' + url.split('js/')[1];
             } else if (url.includes('bundles/')) {
                 caminhoReal = 'bundles/' + url.split('bundles/')[1];
             } else if (url.includes('/')) {
                 caminhoReal = url.substring(url.lastIndexOf('/') + 1);
             }

             url = URL_BASE_JOGO + caminhoReal;
          }

          if (url.endsWith('.png')) {
              url = url.replace('.png', '.webp');
          }

          if (url.includes('lang-en.json') && idioma === 'pt') {
            url = url.replace('lang-en.json', 'lang-pt-br.json');
          }

          args[0] = url;
        }
        return originalFetch.apply(this, args);
      };

      const liberarAudio = () => {
          if (typeof AudioContext !== 'undefined' || typeof webkitAudioContext !== 'undefined') {

                  window.audioContext.resume();
              }
          }
          window.removeEventListener('click', liberarAudio);
          window.removeEventListener('touchstart', liberarAudio);
      };
      window.addEventListener('click', liberarAudio);
      window.addEventListener('touchstart', liberarAudio);

      document.body.appendChild(scriptBoot);
    }
